        audit = ComplianceAudit(req.url)

        audit_counter.labels(status="success").inc()
        audit_duration.observe(time.time() - start)
        audit_score.observe(report.final_score)
        
        return {...}
    except Exception as e:
        audit_counter.labels(status="error").inc()
        raise
