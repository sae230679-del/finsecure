      const steps = 60; // –¥–ª—è –ø–ª–∞–≤–Ω–æ—Å—Ç–∏
      let currentStep = 0;

      const animate = () => {
        currentStep++;
        const progress = currentStep / steps;
        const easeValue = progress < 0.5 
          ? 2 * progress * progress 
          : -1 + (4 - 2 * progress) * progress;
        
        setDisplayValue(start + difference * easeValue);
        
        if (currentStep < steps) {
          animationFrame = requestAnimationFrame(animate);
        }
      };

      animationFrame = requestAnimationFrame(animate);
      return () => cancelAnimationFrame(animationFrame);
    } else {
      setDisplayValue(value);
    }
  }, [value, animated]);

  const statusClass = status === 'auto' ? currentStatus : status;
  const currentTheme = theme === 'auto' 
    ? (window.matchMedia('(prefers-color-scheme: dark)').matches ? 'dark' : 'light')
    : theme;

  const defaultBadgeText = {
    success: '‚úì –í—Å—ë —Ö–æ—Ä–æ—à–æ',
    warning: '‚ö† –¢—Ä–µ–±—É–µ—Ç –≤–Ω–∏–º–∞–Ω–∏—è',
    critical: '‚úï –ö—Ä–∏—Ç–∏—á–Ω–æ',
  };

  return (
    <div className={`progress-wrapper ${className} theme-${currentTheme}`}>
      {label && <label className="progress-label">{label}</label>}
      
      <div 
        className={`progress-container size-${size}`}
        onMouseEnter={(e) => tooltip?.enabled && console.log('tooltip')}
        role="progressbar"
        aria-valuenow={Math.round(displayValue)}
        aria-valuemin="0"
        aria-valuemax="100"
        aria-label={ariaLabel}
      >
        <div className={`progress-bar-track status-${statusClass}`}>
          <div 
            className={`progress-bar-fill`}
            style={{ width: `${displayValue}%` }}
          />
        </div>
        
        {showPercent && (
          <span className="progress-percent">
            {Math.round(displayValue)}%
          </span>
        )}
      </div>
      
      {showBadge && (
        <p className={`progress-badge status-${statusClass}`}>
          {badgeText || defaultBadgeText[statusClass]}
        </p>
      )}
    </div>
  );
};

export default ProgressBar;

class ProgressBar {
  constructor(selector, options = {}) {
    this.element = document.querySelector(selector);
    this.options = {
      value: 0,
      status: 'auto',
      animated: true,
      ...options
    };

    this.createElements();
    this.setProgress(this.options.value);
  }

  createElements() {
    this.element.innerHTML = `
      <div class="progress-bar-track">
        <div class="progress-bar-fill"></div>
      </div>
      <span class="progress-percent">0%</span>
      <p class="progress-badge"></p>
    `;
  }

  setProgress(value) {
    const fill = this.element.querySelector('.progress-bar-fill');
    const percent = this.element.querySelector('.progress-percent');
    
    fill.style.width = value + '%';
    percent.textContent = Math.round(value) + '%';
    
    this.updateStatus(value);
  }

  updateStatus(value) {
    const status = value <= 30 ? 'success' : value <= 70 ? 'warning' : 'critical';
    const badge = this.element.querySelector('.progress-badge');
    badge.className = `progress-badge status-${status}`;
  }
}
